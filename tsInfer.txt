    Arr extends Array<infer U> ? U : never //

type GetSecndParam<F> = 
F extends (p1: any, p2: infer Param, ...args: any[]) => any ? Param : never//o infer vai inferir o tipo do segundo parametro da função, e retornar ele, se não for uma função, vai retornar never 

type funcParam2 = GetSecndParam<typeof createUser>
//--------------------------------------------------------------------

type GetParameters<F> =
F extends (...args: infer Args) => any ? Args : never;

type FuncParams = GetParameters<typeof createUser>//agora o resultado virou uma tupla, com todos os parametros da função, e não apenas o segundo parametro, como no exemplo anterior
